Steps to setup typescript project
1. npm init -y
2. npm install --save-dev typescript
3. npx tsc --init
4. configure ts project according to your need
5. npx tsc
6. npm install --save-dev eslint @typescript-eslint/parser @typescript-eslint/eslint-plugin
7. npm install --save-dev ts-node
8. npm ts-node src/index.ts
9. npm install npm-run-all
10. npx npm-run-all --version
11. npm start
